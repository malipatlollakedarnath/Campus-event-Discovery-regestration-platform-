// Firestore Security Rules
// Apply these rules in Firebase Console -> Firestore Database -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Students Collection
    match /students/{uid} {
      allow read: if request.auth != null && request.auth.uid == uid;
      allow create: if request.auth != null && request.auth.uid == uid;
      allow update: if request.auth != null && request.auth.uid == uid;
      allow delete: if request.auth != null && request.auth.uid == uid;
    }

    // Events Collection - Public Read, Admin Write
    match /events/{document=**} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && isAdmin();
      allow update: if request.auth != null && isAdmin();
      allow delete: if request.auth != null && isAdmin();
    }

    // Registrations Collection
    match /registrations/{document=**} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.studentId;
      allow update: if request.auth != null && request.auth.uid == resource.data.studentId;
      allow delete: if request.auth != null && (request.auth.uid == resource.data.studentId || isAdmin());
    }

    // Helper function to check if user is admin
    function isAdmin() {
      return request.auth != null &&
             (request.auth.token.email.matches('.*@admin$') ||
              request.auth.token.email.matches('.*admin@college.*'));
    }
  }
}

/*
IMPORTANT: Copy and paste the rules above into Firebase Console:
1. Go to Firebase Console -> Your Project
2. Select "Firestore Database" from left menu
3. Click "Rules" tab
4. Replace the default rules with the above
5. Click "Publish"

For development/testing, you can use these simpler rules (NOT recommended for production):

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
